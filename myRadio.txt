// 🔁 Convert YouTube link → embed
function convertToEmbed(url) {
  let id = "";
  if (url.includes("watch?v=")) {
    id = url.split("watch?v=")[1].split("&")[0];
  } else if (url.includes("youtu.be/")) {
    id = url.split("youtu.be/")[1];
  }
  return "https://www.youtube.com/embed/" + id;
}

// 📺 CHANNEL LIST
const channels = [
  {
    channel: 1,
    name: "News TV",
    url: "https://youtu.be/YznJM0k4Bok?si=yHbjhU2Lykc25bKt"
  },
  {
    channel: 15,
    name: "Music TV",
    url: "https://www.youtube.com/watch?v=jfKfPfyJRdk"
  },
  {
    channel: 30,
    name: "Pop Hits",
    url: "https://www.youtube.com/watch?v=kJQP7kiw5Fk"
  },
  {
    channel: 60,
    name: "Rock Channel",
    url: "https://www.youtube.com/watch?v=ktvTqknDobU"
  }
];

// ELEMENTS
const tuner = document.getElementById("tuner");
const display = document.getElementById("channelDisplay");
const status = document.getElementById("status");
const player = document.getElementById("player");

let current = null;

// 🎚 TUNER
tuner.addEventListener("input", () => {
  const ch = parseInt(tuner.value);
  display.innerText = "Channel " + ch;

  let found = channels.find(c => c.channel === ch);

  if (found) {
    tune(found);
  } else {
    noSignal();
  }
});

// 📺 TUNE CHANNEL
function tune(ch) {
  if (current === ch) return;

  current = ch;
  status.innerText = "Now Playing: " + ch.name;

  const embed = convertToEmbed(ch.url);
  player.src = embed + "?autoplay=1";
}

// ❌ NO SIGNAL
function noSignal() {
  current = null;
  player.src = "";
  status.innerText = "No Signal";
}
